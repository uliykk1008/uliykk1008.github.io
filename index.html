void main() {
  const i = 20;

  print('fibonacci($i) = ${fibonacci(i)}');
  print('ULIYA GUMPARI');
}

int fibonacci(int n) {
  return n < 2 ? n : (fibonacci(n - 1) + fibonacci(n - 2));
}
