// main.dart
void main() {
  const i = 20;

  print('fibonacci($i) = ${fibonacci(i)}');
  print('ULIYA GUMPARI');
}

/// Computes the nth Fibonacci number.
int fibonacci(int n) {
  // Perbaikan: harus n-2 (bukan -2)
  return n < 2 ? n : (fibonacci(n - 1) + fibonacci(n - 2));
}
